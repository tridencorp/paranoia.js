class Transaction {
  constructor(attrs) {
    this.nonce  = Uint32;
    this.from   = new Uint8(20); // 20 bytes
    this.to     = new Uint8(20); // 20 bytes
    this.gas    = Uint32;
    this.amount = Uint32;
  }
}

let tx = new Transaction()

tx.nonce  = new Uint32([1_000])
tx.from   = new Uint8([111, 143, 87, 113, 80, 144, 218, 38, 50, 69, 57, 136, 217, 161, 80, 27, 154, 250, 29, 11])
